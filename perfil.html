// ==================== PERFIL (con botón editar/eliminar) ==================== function cargarPerfil() { const usuarioActual = getUsuarioActual(); if (!usuarioActual) { window.location.href = 'index.html'; return; } document.getElementById('nombreUsuario').textContent = usuarioActual.nombre; document.getElementById('emailUsuario').textContent = usuarioActual.email; const productos = getProductos(); const misProductos = productos.filter(p => p.vendedor === usuarioActual.nombre); const contenedor = document.getElementById('misPublicaciones'); if (misProductos.length === 0) { contenedor.innerHTML = `

📭 No tienes publicaciones

Subí tu primera camiseta

`; return; } contenedor.innerHTML = misProductos.map(producto => `
👕
${producto.nombre}
$${producto.precio.toLocaleString()}
${producto.etiquetas ? producto.etiquetas.map(tag => `${tag}`).join('') : ''}
${producto.disponible ? 'Disponible' : 'Vendido'}
Ver
`).join(''); // Agregar eventos a los botones de editar en el perfil document.querySelectorAll('.btn-editar-perfil').forEach(btn => { btn.addEventListener('click', () => { const id = parseInt(btn.getAttribute('data-id')); window.location.href = `publicacion.html?id=${id}`; }); }); }
